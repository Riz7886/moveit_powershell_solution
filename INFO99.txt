$subId = az account show --query id -o tsv
$token = az account get-access-token --query accessToken -o tsv

$url = "https://management.azure.com/subscriptions/$subId/providers/Microsoft.Capacity/resourceProviders/Microsoft.Compute/locations/westus/serviceLimitsRequests/standardEDSv4Family?api-version=2020-10-25"

$body = @{
    properties = @{
        requestedResource = @{
            limit = 64
            name = @{ value = "standardEDSv4Family" }
            resourceType = "dedicated"
        }
    }
} | ConvertTo-Json -Depth 10

Invoke-RestMethod -Uri $url -Method Put -Headers @{Authorization="Bearer $token";"Content-Type"="application/json"} -Body $body
